ACTOR KickWeapon : Weapon
{
	States
	{
	Select:
		TNT1 A 1 A_Raise
		Wait
	Deselect:
		TNT1 A 1 A_Lower
		Wait
	Ready:
        TNT1 A 0 A_ZoomFactor(1.0)
		TNT1 A 0 A_JumpIf(momZ > 0, "AirKick")
		TNT1 A 0 A_JumpIf(momZ < 0, "AirKick")
        TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, "BerserkerKick")
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 A_AlertMonsters (500, 0)
		KICK BCD 1
		TNT1 A 0 A_FireCustomMissile("KickAttack", 0, 0, 0, -7)
        KICK H 2 
		KICK IGFEDCB 2
		TNT1 A 0 
		KICK A 0 A_TakeInventory("Kicking", 1)
		Goto Ready
	BerserkerKick:
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 A_AlertMonsters (500, 0)
		KICK BCD 1
        TNT1 A 0 A_FireCustomMissile("SuperKickAttack", 0, 0, 0, -7)
        KICK H 2
		KICK IGFEDCB 2
		TNT1 A 0 SetPlayerProperty(0, 0, 0)
		KICK A 0 A_TakeInventory("Kicking", 1)
		Goto InvisLoop
	AirKick:
	    TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, "SuperAirKick")
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 A_Recoil(-6)
		KICK JKLMN 1
        TNT1 A 0 A_FireCustomMissile("AirKickAttack", 0, 0, 0, -31)
        KICK O 2 A_AlertMonsters (500, 0)
		KICK PQRST 2
		KICK A 0 A_TakeInventory("Kicking", 1)
		Goto InvisLoop
	SuperAirKick:
		TNT1 A 0 A_PlaySound("KICK")
		TNT1 A 0 A_Recoil(-6)
		KICK JKLMN 1
        TNT1 A 0 A_FireCustomMissile("SuperAirKickAttack", 0, 0, 0, -31)
        KICK O 2 A_AlertMonsters (500, 0)
		KICK PQRST 2
		KICK A 0 A_TakeInventory("Kicking",1)
		Goto InvisLoop
	InvisLoop:
	Fire:
		TNT1 A 1
		Loop
	}
}

ACTOR Kicking : Inventory
{
	Inventory.MaxAmount 1
}
