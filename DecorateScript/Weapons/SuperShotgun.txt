actor DESuperShotgun : SuperShotgun replaces SuperShotgun
{
  Weapon.BobStyle smooth
  States
  {
  Select:
	SHT2 A 0 A_SetCrossHair(10)
	SHT2 A 1 A_Raise
    Loop
  Fire:
    SHT2 A 7
	{
		A_WeaponOffset(0, 42, WOF_KEEPX | WOF_INTERPOLATE);
		A_QuakeEx(0, 0, 1, 2, 0, 1, "", QF_RELATIVE);
		A_FireShotgun2;
	}
    SHT2 B 7 A_WeaponOffset(0, 32, WOF_KEEPX | WOF_INTERPOLATE)
    SHT2 C 7 A_CheckReload
    SHT2 D 7 A_OpenShotgun2
    SHT2 E 7
    SHT2 F 7 A_LoadShotgun2
    SHT2 G 6
    SHT2 H 6 A_CloseShotgun2
    SHT2 A 5 A_ReFire
    Goto Ready
    // unused states
    SHT2 B 7
    SHT2 A 3
    Goto Deselect
  Flash:
    SHT2 I 1 Bright A_Light1
	SHT2 J 3 Bright A_Light2
    SHT2 I 2 Bright A_Light1
    Goto LightDone
  }
}
