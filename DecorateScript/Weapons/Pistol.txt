ACTOR DEPistol : Pistol replaces Pistol
{
  Weapon.BobStyle smooth
  States
  {
  Select:
    PISG A 0 A_SetCrosshair(8)
	PISG A 1 A_Raise
	Loop
  Fire:
	PISG B 3
	PISG B 6 
	{
		A_WeaponOffset(0, 30, WOF_KEEPX | WOF_INTERPOLATE);
		A_FirePistol;
	}
	PISG C 5 A_WeaponOffset(0, 29, WOF_KEEPX | WOF_INTERPOLATE)
	PISG B 5 
	{
		A_WeaponOffset(0, 32, WOF_KEEPX | WOF_INTERPOLATE);
		A_ReFire;
	}
	Goto Ready
  Flash:
	PISF A 4 Bright A_Light1
	Goto LightDone
	TNT1 A 3 Bright A_Light1
	Goto LightDone
  }
}
