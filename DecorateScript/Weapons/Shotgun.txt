actor DEShotgun : Shotgun replaces Shotgun
{
  Weapon.BobStyle smooth
  States
  {
  Select:
	SHTG A 0 A_SetCrossHair(9)
	SHTG A 1 A_Raise
    Loop
  Fire:
    SHTG A 7
	{
		A_WeaponOffset(0, 40, WOF_KEEPX | WOF_INTERPOLATE);
		A_FireShotgun;
	}
    SHTG BC 2 A_WeaponOffset(0, 32, WOF_KEEPX | WOF_INTERPOLATE)
    SHTG DEFGHIJ 1
    SHTG IH 4
	SHTG GFEDCB 1
    SHTG A 3
    SHTG A 7 A_ReFire
    Goto Ready
  Flash:
    SHTF A 1 Bright A_Light1
	SHTF B 3 Bright A_Light2
    SHTF A 2 Bright A_Light1
    Goto LightDone
  }
} //25 24
