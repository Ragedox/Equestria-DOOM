actor DEShotgun : Shotgun replaces Shotgun
{
  Weapon.BobStyle smooth
  States
  {
  Select:
	SHTG A 0 A_SetCrossHair(9)
	SHTG A 1 A_Raise
    Loop
  Fire:
    SHTG A 3
    SHTG A 7 
	{
		A_WeaponOffset(0, 40, WOF_KEEPX | WOF_INTERPOLATE);
		A_FireShotgun;
	}
    SHTG BC 5 A_WeaponOffset(0, 32, WOF_KEEPX | WOF_INTERPOLATE)
    SHTG D 4
    SHTG CB 5
    SHTG A 3
    SHTG A 7 A_ReFire
    Goto Ready
  Flash:
    SHTF B 4 Bright A_Light1
    TNT1 A 3 Bright A_Light2
    Goto LightDone
  }
}
