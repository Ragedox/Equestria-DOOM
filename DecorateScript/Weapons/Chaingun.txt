actor DEChaingun : Chaingun replaces Chaingun
{
  Weapon.BobStyle alpha
  Inventory.PickupSound "pickup/chaingun"
  States
  {
  Select:
	CHGG A 0 A_SetCrossHair(11)
	CHGG A 1 A_Raise
    Loop
  Fire:
	CHGG AB 4 
	{
		A_FireCGun;
		A_QuakeEx(1, 0, 1, 4, 0, 1, "", QF_RELATIVE | QF_SCALEDOWN | QF_SCALEUP);
	}
	CHGG B 0 A_ReFire
    Goto Ready
  Flash:
    CHGF A 2 Bright A_WeaponOffset(0, 38, WOF_KEEPX | WOF_INTERPOLATE)
    Goto LightDone
	TNT1 A 3 Bright 
	CHGF B 2 Bright A_WeaponOffset(0, 40, WOF_KEEPX | WOF_INTERPOLATE)
    Goto LightDone
  }
}
