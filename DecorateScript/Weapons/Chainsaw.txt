actor DEChainsaw : Chainsaw replaces Chainsaw
{
  States
	{
  Ready:
	SAWG CD 4 A_WeaponReady
	Loop
  Fire:
	SAWG A 2 A_WeaponOffset(random(0,1),random(38, 40), WOF_INTERPOLATE)
	SAWG B 2
	{
		A_WeaponOffset(random(-1,0),random(38, 36), WOF_INTERPOLATE);
		A_Saw;
	}
	SAWG B 0 A_ReFire
	Goto Ready
	}
}
