actor DEChainsaw : Chainsaw replaces Chainsaw
{
  Weapon.UpSound ""
  Weapon.ReadySound "weapons/sawidle"
  Weapon.BobStyle Smooth
  States
	{
  Select:
	TNT1 A 0 A_StartSound("weapons/sawup")
	SAWG A 1 A_Raise
	Loop
  Deselect:
    SAWG B 1 A_Lower
    Loop
  Ready:
	SAWG C 2 
	{
		A_WeaponOffset(frandom(-1,1), frandom(30, 40), WOF_INTERPOLATE);
		A_WeaponReady;
	}
	SAWG D 2 
	{
		A_WeaponOffset(frandom(-1,1), frandom(30, 40), WOF_INTERPOLATE);
		A_WeaponReady;
	}
	Loop
  Fire:
	SAWG A 1 A_WeaponOffset(frandom(-1,1),frandom(30, 32), WOF_INTERPOLATE)
	SAWG B 1 A_WeaponOffset(frandom(-1,1),frandom(30, 32), WOF_INTERPOLATE)
	SAWG A 2
	{
		A_WeaponOffset(frandom(-1,1),frandom(30, 32), WOF_INTERPOLATE);
		A_Saw;
	}
	SAWG B 0 A_ReFire
	Goto Ready
	}
}
