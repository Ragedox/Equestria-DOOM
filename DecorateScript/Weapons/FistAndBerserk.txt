actor DEBerserk : Berserk replaces Berserk
{
  Inventory.PickupSound "*xdeath"
  States
  {
  Pickup:
	TNT1 A 0 A_GiveInventory("PowerStrength")
    TNT1 A 0 HealThing(100, 0)
    TNT1 A 0 A_SelectWeapon("Fist")
	TNT1 A 1 A_SpawnItemEx("BerserkTransform", 0, 0, 28, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	Stop
  }
}

actor BerserkTransform
{
  Monster
  Mass 999999
  RenderStyle add
  Alpha 0.75
  MaxTargetRange 999999
  Speed 0
  +FLOORCLIP
  States
  {
  Spawn:
	VILE A 0 A_LookEx(LOF_NOSOUNDCHECK, 0, 1280, 0, 360, "Missile")
	Loop
  See:
	VILE A 0
	Loop
  Missile:
	VILE G 0 BRIGHT A_VileStart
	VILE G 10 BRIGHT A_FaceTarget
	VILE H 8 BRIGHT A_VileTarget
	VILE IJKLMN 8 BRIGHT A_FaceTarget
	VILE O 8 BRIGHT
	Goto Death
  Death:
    VILE A 0
	Stop
  }
}
