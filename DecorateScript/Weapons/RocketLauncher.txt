actor DERocketLauncher : RocketLauncher replaces RocketLauncher
{
  Weapon.BobStyle alpha
  States
  {
  Select:
	MISG A 0 A_SetCrossHair(13)
	MISG A 1 A_Raise
	Loop
  Fire:
    MISG B 8 A_GunFlash
    MISG B 12 
	{
		A_WeaponOffset(0, 40, WOF_KEEPX | WOF_INTERPOLATE);
		A_FireMissile;
	}
    MISG B 0
	{
		A_WeaponOffset(0, 32, WOF_KEEPX | WOF_INTERPOLATE);
		A_ReFire;
	}
    Goto Ready
  }
}
