actor EquestriaCacodemon : Cacodemon replaces Cacodemon
{
	BloodColor "blue"
	
	
	+CASTSPRITESHADOW
	
	+FLOATBOB
	FloatBobStrength 0.5
	
	+MISSILEMORE
	+FORCEXYBILLBOARD
	
	MeleeThreshold 100
	
	States
	{

	Melee:
		HEAD BCDD 2 A_FaceTarget
		HEAD C 3
		{
			A_PlaySound();
			A_FaceTarget;
		}
		HEAD C 3 A_Recoil(-15)
		HEAD B 4 A_CustomMeleeAttack(10 * random(1, 6), "demon/melee", "skeleton/swing")
		
		HEAD B 4 A_Stop
		Goto See
	
	Missile:
		HEAD BC 5 A_FaceTarget
		HEAD D 5 Bright A_CustomMissile("CacodemonBall", 32)
		Goto See
	
	Death:
		TNT1 A 0 
		{
			A_Scream;
			bFLOATBOB = 0;
			bFORCEXYBILLBOARD = 0;
		}
		HEAD G 4 A_SetSolid
		HEAD G 4 A_SetShootable
		HEAD H 8 
		HEAD H -1
		Stop
	
	Crash:
		HEAD IJ 4
		HEAD K 2 A_NoBlocking
		HEAD K 1 A_UnsetSolid
		HEAD K 1 A_UnSetShootable
		HEAD L -1 A_SetFloorClip
		Stop
	
	Raise:
		HEAD L 8 A_UnSetFloorClip
		HEAD KJIHG 8
		
		HEAD A 0 
		{
			bFLOATBOB = 1;
			bFORCEXYBILLBOARD = 1;
		}
		Goto See

	}
}
