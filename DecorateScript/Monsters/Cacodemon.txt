actor EquestriaCacodemon : Cacodemon replaces Cacodemon
{
	BloodColor "blue"
	FloatBobStrength 0.5
	
	+CASTSPRITESHADOW
	+FLOATBOB
	+MISSILEMORE
	+FORCEXYBILLBOARD
	
	States
	{
	
	FallBack:
		HEAD A 4 A_Recoil(4)
		HEAD A 0 A_Stop
		Goto See
		
	Melee:
		HEAD BC 2 A_Chase
		HEAD D 3 A_Chase
		HEAD DC 3 A_Chase
		{
			A_PlaySound("demon/melee");
			A_FaceTarget;
		}
		HEAD C 3 A_Recoil(-10)
		HEAD B 4 A_CustomMeleeAttack(10 * random(1, 6))
		
		HEAD B 0 A_Stop
		
		HEAD B 0 A_JumpIfCloser(50, "FallBack")
		Goto See
	
	Missile:
		HEAD D 0 A_JumpIfCloser(128, "Melee")
		HEAD BC 5 A_FaceTarget
		HEAD D 5 Bright A_CustomMissile("CacodemonBall", 32)
		Goto See
	
	Death:
		TNT1 A 0 A_Changeflag(FLOATBOB,0)
		HEAD G 4 A_SetSolid
		HEAD G 4 A_SetShootable
		HEAD H 8 A_Scream
		HEAD H -1
		Stop
	
	Crash:
		HEAD I 4 A_PlaySound ("*fist")
		HEAD J 4
		HEAD K 2 A_NoBlocking
		HEAD K 1 A_UnsetSolid
		HEAD K 1 A_UnSetShootable
		HEAD L -1 A_SetFloorClip
		Stop
	
	Raise:
		HEAD L 8 A_UnSetFloorClip
		HEAD KJIHG 8
		Goto See

	}
}
