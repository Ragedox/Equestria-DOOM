Actor FloorCheck
{
	Radius 1
	Height 1
	+MISSILE
	+NOTELEPORT
	+NOBLOCKMAP
	+THRUACTORS
	+MOVEWITHSECTOR
	+NODAMAGETHRUST	
	+FLOORCLIP
	+GHOST
	+NOGRAVITY
}

Actor BulletFloorCheck : FloorCheck
{
	States
	{
	
	Spawn:
		TNT1 A 0
		TNT1 A 0 A_CheckFloor("DecalFloor")
		TNT1 A 0 A_CheckCeiling("DecalCeiling")
		Stop
	
	DecalFloor:
		TNT1 A 0
		TNT1 A 0 A_SpawnItemEX("BulletDecal")
		Stop
	
	DecalCeiling:
		TNT1 A 0
		TNT1 A 0 A_SpawnItemEX("BulletDecal")
		Stop
	}
}
