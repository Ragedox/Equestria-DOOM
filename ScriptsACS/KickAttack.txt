#library "KickAttack"
#include "zcommon.acs"

Script "GiveKick" ENTER
{
    If(!CheckInventory("QuickMelee")){GiveInventory("QuickMelee",1);} //Give the player the item, not needed if the custom class starts with it
    While(true)
    {
        If(GetPlayerInput(-1,INPUT_OLDBUTTONS)&BT_RELOAD) //Replace Reload with any key (BT_ALTATTACK / BT_ZOOM / BT_USER1 - BT_USER4)
        {UseInventory("QuickMelee");}
        Delay(1);
    }
}
