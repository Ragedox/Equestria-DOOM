//water splash
splash DoomWater
{
   smallclass     	WaterSplashBase
   smallclip		0
   smallsound     	world/drip

   baseclass      	WaterSplashBase
   chunkclass     	WaterSplash
   sound          	world/watersplash
   noalert
}

terrain DoomWater
{
	splash			DoomWater
	footclip		6
	liquid
}
